pie_et_sgui_init = {
	scope = country
	effect = {
		pie_et_init_trade = yes
	}
}

pie_et_sgui_open_window = {
	scope = country
	effect = {
		pie_et_init_trade = yes
		pie_et_queue_trade_mapmode = yes
		set_variable = { name = pie_et_window_open value = yes }
	}
}

pie_et_sgui_start_proxy_trade = {
	scope = country
	is_valid = {
		OR = {
			country_has_estate = estate_type:nobles_estate
			country_has_estate = estate_type:clergy_estate
			country_has_estate = estate_type:burghers_estate
		}
		NOT = { at_war = yes }
	}
	effect = {
		pie_et_open_trade_choice_event = yes
	}
}

pie_et_sgui_close_window = {
	scope = country
	effect = {
		pie_et_clear_selected_land_locations = yes
		remove_variable = pie_et_window_open
		remove_variable = pie_et_estate_picker_open
		remove_variable = pie_et_land_sell_open
		remove_variable = pie_et_land_buy_open
		clear_variable_list = pie_et_sell_land_locations
		clear_variable_list = pie_et_buy_land_locations
		clear_variable_list = pie_et_mapmode_trigger
	}
}
