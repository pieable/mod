pie_ab_sgui_compare_new_profit = {
	scope = location
	saved_scopes = { bt profit cost }
	effect = {
		if = {
			limit = {
				scope:profit > var:pie_ab_best_profit
				can_build_building = scope:bt
				OR = {
					NOT = { owner = { has_variable = pie_ab_profit_threshold } }
					scope:profit >= owner.var:pie_ab_profit_threshold
				}
				OR = {
					NOT = { has_building = scope:bt }
					"building(scope:bt|owner)" = {
						pie_ab_remaining_build_capacity > 0
					}
				}
			}
			set_variable = { name = pie_ab_best_profit value = scope:profit }
			set_variable = { name = pie_ab_best_bt value = scope:bt }
			set_variable = { name = pie_ab_best_cost value = scope:cost }
		}
	}
}

	}
}

pie_ab_sgui_precompute_all = {
	scope = country
	effect = {
		if = {
			limit = { NOT = { has_variable = pie_ab_profit_threshold } }
			set_variable = { name = pie_ab_profit_threshold value = 0 }
		}
		if = {
			limit = { NOT = { has_variable = pie_ab_gold_limit } }
			set_variable = { name = pie_ab_gold_limit value = 0 }
		}
		pie_ab_queue_precompute_enabled_locations = yes
	}
}

pie_ab_sgui_toggle_all_auto_build = {
	scope = country
	effect = {
		if = {
			limit = {
				any_owned_location = {
					has_variable = pie_ab_auto_build
				}
			}
			every_owned_location = {
				limit = { has_variable = pie_ab_auto_build }
				remove_variable = pie_ab_auto_build
			}
		}
		else = {
			every_owned_location = {
				set_variable = { name = pie_ab_auto_build value = 1 }
			}
		}
		pie_ab_queue_precompute_enabled_locations = yes
	}
}

pie_ab_sgui_toggle_core_auto_build = {
	scope = country
	effect = {
		if = {
			limit = {
				any_owned_location = {
					has_variable = pie_ab_auto_build

				}
			}
			every_owned_location = {
				limit = {
					has_variable = pie_ab_auto_build
					integration_level = core
				}

			}
		}
		pie_ab_queue_precompute_enabled_locations = yes
	}
}
