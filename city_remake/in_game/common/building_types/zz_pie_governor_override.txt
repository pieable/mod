REPLACE:local_governor = {
	is_foreign = no
	pop_type = nobles
	max_levels = 1
	important_for_UI = yes

	employment_size = large_noble_employment

	category = government_category

	rural_settlement = yes
	town = yes
	city = yes

	allow = {
		custom_tooltip = {
			text = local_governor_tt
			scope:actor = { total_building_levels_including_construction:local_governor < modifier:num_local_governors }
		}
		owner = scope:actor
	}

	location_potential = {
		has_road_to_capital = yes
		owner.capital != this
	}

	remove_if = {
		location = { has_road_to_capital = no }
	}

	expensive = yes
	price = pie_governor_building

	automation_build_allowed = no
	ai_forbid_shutdown = yes
	forbidden_for_estates = yes

	build_time = government_build_time

	unique_production_methods = {
		local_governor_maintenance = {
			tools = 1
			masonry = 1
			weaponry = 0.5
			paper = 3
			books = 1
			goods_gold = 1
			furniture = 2
			fine_cloth = 1

			category = building_maintenance
		}
	}

	construction_demand = local_governor_construction

	modifier = {
		local_proximity_source = 90
		local_monthly_development = 0.005
		local_migration_attraction = 0.1
	}
}
