REPLACE:stockade = {
	max_levels = 1
	pop_type = soldiers

	category = defense_category

	employment_size = stockade_employment

	rural_settlement = yes
	town = yes
	city = yes

	build_time = small_fort_building

	expensive = yes

	unique_production_methods = {
		pie_stockade_maintenance = {
			lumber = 0.1
			weaponry = 0.02
			tar = 0.02
			category = building_maintenance
		}
	}
	construction_demand = stockade_construction

	allow = {
		NOR = {
			has_building = building_type:castle
			has_building = building_type:bastion
			has_building = building_type:star_fort
			has_building = building_type:fortress
		}
	}

	on_built = {
		location = {
			if = {
				limit = {
					OR = {
						has_building = building_type:castle
						has_building = building_type:bastion
						has_building = building_type:star_fort
						has_building = building_type:fortress
					}
				}
				every_buildings_in_location = {
					limit = {
						building_type = building_type:stockade
					}
					location = { destroy_building = prev }
				}
			}
		}
	}

	modifier = {
		local_garrison_size = 0.1
		local_unrest = -0.025
		local_nobles_estate_power = 0.5
	}

	raw_modifier = {
		fort_level = 1
		local_nobles_desired_pop = 0.005
		propagating_zone_of_control = no
	}
}
