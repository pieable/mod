pie_et_has_selected_estate = {
	OR = {
		AND = {
			has_variable = pie_et_estate_nobles
			country_has_estate = estate_type:nobles_estate
		}
		AND = {
			has_variable = pie_et_estate_clergy
			country_has_estate = estate_type:clergy_estate
		}
		AND = {
			has_variable = pie_et_estate_burghers
			country_has_estate = estate_type:burghers_estate
		}
	}
}

pie_et_debt_slot_is_open = {
	OR = {
		NOT = { has_variable = $debt_var$ }
		AND = {
			has_variable = $debt_var$
			var:$debt_var$ <= 0
		}
	}
}

pie_et_has_any_trade_term = {
	OR = {
		AND = {
			has_variable = pie_et_gold
			var:pie_et_gold > 0
		}
		AND = {
			has_variable = pie_et_gold
			var:pie_et_gold < 0
		}
		has_variable = pie_et_term_royal_credit_pledge
		has_variable = pie_et_term_estate_funds_project
		has_variable = pie_et_term_crown_grants_rights
		pie_et_has_selected_privilege_term = yes
		pie_et_has_selected_sell_land = yes
		pie_et_has_selected_grant_unregistered_land = yes
		pie_et_has_selected_buy_land = yes
		AND = {
			pie_et_has_selected_project = yes
			pie_et_has_project_receiver = yes
		}
	}
}

pie_et_has_selected_privilege_term = {
	OR = {
		pie_et_generated_selected_privilege_is_valid = yes
		pie_et_generated_peasant_selected_privilege_is_valid = yes
		has_variable = pie_et_priv_grant_nobles_land_rights
		has_variable = pie_et_priv_revoke_nobles_land_rights
		has_variable = pie_et_priv_grant_nobility_council
		has_variable = pie_et_priv_revoke_nobility_council
		has_variable = pie_et_priv_grant_clerical_advisory_council
		has_variable = pie_et_priv_revoke_clerical_advisory_council
		has_variable = pie_et_priv_grant_clergy_religious_state
		has_variable = pie_et_priv_revoke_clergy_religious_state
		has_variable = pie_et_priv_grant_formal_guilds
		has_variable = pie_et_priv_revoke_formal_guilds
		has_variable = pie_et_priv_grant_treasury_rights
		has_variable = pie_et_priv_revoke_treasury_rights
	}
}

pie_et_selected_estate_can_lend_gold = {
	OR = {
		AND = {
			has_variable = pie_et_estate_nobles
			"estate(estate_type:nobles_estate)" = { estate_gold >= root.var:pie_et_gold }
		}
		AND = {
			has_variable = pie_et_estate_clergy
			"estate(estate_type:clergy_estate)" = { estate_gold >= root.var:pie_et_gold }
		}
		AND = {
			has_variable = pie_et_estate_burghers
			"estate(estate_type:burghers_estate)" = { estate_gold >= root.var:pie_et_gold }
		}
	}
}

				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_nobles_6 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_nobles_7 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_nobles_8 }
			}
		}
		AND = {

			OR = {
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_1 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_2 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_3 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_4 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_5 }

				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_7 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_clergy_8 }
			}
		}
		AND = {
			has_variable = pie_et_estate_burghers
			OR = {
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_burghers_1 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_burghers_2 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_burghers_3 }
				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_burghers_4 }

				pie_et_debt_slot_is_open = { debt_var = pie_et_debt_burghers_8 }

		}

pie_et_loan_within_country_capacity = {
	country_loan_capacity >= pie_et_total_estate_debt_after_trade
}

pie_et_selected_estate_land_grant_candidate = {
	OR = {
		AND = {
			has_variable = pie_et_estate_nobles
			any_owned_location = { pie_land_nobles_grant_candidate = yes }
		}
		AND = {
			has_variable = pie_et_estate_clergy
			any_owned_location = { pie_land_clergy_grant_candidate = yes }
		}
		AND = {
			has_variable = pie_et_estate_burghers
			any_owned_location = { pie_land_burghers_grant_candidate = yes }
		}
	}
}

pie_et_sell_land_candidate = {
	is_ownable = yes
	has_owner = yes
	is_land = yes
	NOT = { is_capital = yes }
	has_siege = no
	is_looted = no
	has_variable = pie_land_owner_crown
	owner ?= { pie_et_has_selected_estate = yes }
}

pie_et_grant_unregistered_land_candidate = {
	is_ownable = yes
	has_owner = yes
	is_land = yes
	NOT = { is_capital = yes }
	has_siege = no
	is_looted = no
	NOT = { pie_land_has_defined_owner = yes }
	owner ?= { pie_et_has_selected_estate = yes }
}

pie_et_buy_land_candidate = {
	is_ownable = yes
	has_owner = yes
	is_land = yes
	NOT = { is_capital = yes }
	has_siege = no
	is_looted = no
	owner ?= { pie_et_has_selected_estate = yes }
	OR = {
		AND = {
			owner ?= { has_variable = pie_et_estate_nobles }
			has_variable = pie_land_owner_nobles
		}
		AND = {
			owner ?= { has_variable = pie_et_estate_clergy }
			has_variable = pie_land_owner_clergy
		}
		AND = {
			owner ?= { has_variable = pie_et_estate_burghers }
			has_variable = pie_land_owner_burghers
		}
	}
}

pie_et_selected_estate_project_candidate = {
	is_ownable = yes
	has_owner = yes
	is_land = yes
	owner ?= { pie_et_has_selected_estate = yes }
	OR = {
		has_variable = pie_town_project_active
		has_variable = pie_city_project_active
	}
	OR = {
		AND = {
			owner ?= { has_variable = pie_et_estate_nobles }
			has_variable = pie_project_estate_nobles
		}
		AND = {
			owner ?= { has_variable = pie_et_estate_clergy }
			has_variable = pie_project_estate_clergy
		}
		AND = {
			owner ?= { has_variable = pie_et_estate_burghers }
			has_variable = pie_project_estate_burghers
		}
	}
}

pie_et_has_project_receiver = {
	OR = {
		AND = {
			has_variable = pie_et_receiver_nobles
			country_has_estate = estate_type:nobles_estate
			NOT = { has_variable = pie_et_estate_nobles }
		}
		AND = {
			has_variable = pie_et_receiver_clergy
			country_has_estate = estate_type:clergy_estate
			NOT = { has_variable = pie_et_estate_clergy }
		}
		AND = {
			has_variable = pie_et_receiver_burghers
			country_has_estate = estate_type:burghers_estate
			NOT = { has_variable = pie_et_estate_burghers }
		}
	}
}

pie_et_can_pay_offered_gold = {
	gold >= {
		value = var:pie_et_gold
		multiply = -1
	}
}

pie_et_trade_can_accept = {
	pie_et_has_selected_estate = yes
	pie_et_has_any_trade_term = yes
	pie_et_acceptance_value >= 0
	OR = {
		NOT = {
			AND = {
				has_variable = pie_et_gold
				var:pie_et_gold > 0
			}
		}
		AND = {
			pie_et_selected_estate_can_lend_gold = yes
			pie_et_selected_estate_has_open_debt_slot = yes
			pie_et_loan_within_country_capacity = yes
		}
	}
	OR = {
		NOT = {
			AND = {
				has_variable = pie_et_gold
				var:pie_et_gold < 0
			}
		}
		pie_et_can_pay_offered_gold = yes
	}
	OR = {
		NOT = { has_variable = pie_et_term_crown_grants_rights }
		pie_et_selected_estate_land_grant_candidate = yes
	}
	OR = {
		NOT = { pie_et_has_selected_project = yes }
		pie_et_has_project_receiver = yes
	}
	OR = { NOT = { has_variable = pie_et_priv_grant_nobles_land_rights } pie_et_can_grant_nobles_land_rights = yes }
	OR = { NOT = { has_variable = pie_et_priv_revoke_nobles_land_rights } pie_et_can_revoke_nobles_land_rights = yes }
	OR = { NOT = { has_variable = pie_et_priv_grant_nobility_council } pie_et_can_grant_nobility_council = yes }
	OR = { NOT = { has_variable = pie_et_priv_revoke_nobility_council } pie_et_can_revoke_nobility_council = yes }
	OR = { NOT = { has_variable = pie_et_priv_grant_clerical_advisory_council } pie_et_can_grant_clerical_advisory_council = yes }
	OR = { NOT = { has_variable = pie_et_priv_revoke_clerical_advisory_council } pie_et_can_revoke_clerical_advisory_council = yes }
	OR = { NOT = { has_variable = pie_et_priv_grant_clergy_religious_state } pie_et_can_grant_clergy_religious_state = yes }
	OR = { NOT = { has_variable = pie_et_priv_revoke_clergy_religious_state } pie_et_can_revoke_clergy_religious_state = yes }
	OR = { NOT = { has_variable = pie_et_priv_grant_formal_guilds } pie_et_can_grant_formal_guilds = yes }
	OR = { NOT = { has_variable = pie_et_priv_revoke_formal_guilds } pie_et_can_revoke_formal_guilds = yes }
	OR = { NOT = { has_variable = pie_et_priv_grant_treasury_rights } pie_et_can_grant_treasury_rights = yes }
	OR = { NOT = { has_variable = pie_et_priv_revoke_treasury_rights } pie_et_can_revoke_treasury_rights = yes }
}
