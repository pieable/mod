		has_variable = pie_land_owner_nobles
		has_variable = pie_land_owner_clergy
		has_variable = pie_land_owner_burghers
		has_variable = pie_land_owner_peasants
	}
}

pie_land_natural_close_culture = {
	owner = { has_primary_or_accepted_culture = prev.dominant_culture }
}

pie_land_natural_tolerated_culture = {
	dominant_culture = { is_tolerated_in = prev.owner }
}

pie_land_natural_neutral_culture = {
	dominant_culture = {
		OR = {
			cultural_view = {
				target = prev.owner.culture
				value >= neutral
			}
			reverse_cultural_view = {
				target = prev.owner.culture
				value >= neutral
			}
		}
	}
}

	dominant_religion = owner.religion
}

}
