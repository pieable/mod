pie_pop_current_wealth_extra = {
	value = 0

	if = {
		limit = { pop_type = pop_type:nobles }
		add = {
			value = location.pie_pop_local_noble_wealth_scale
			subtract = 1
			min = 0
			multiply = 0.35
			max = 0.8
		}
	}
	else_if = {
		limit = { pop_type = pop_type:clergy }
		add = {
			value = location.pie_pop_local_clergy_wealth_scale
			subtract = 1
			min = 0
			multiply = 0.45
			max = 0.9
		}
	}
	else_if = {
		limit = { pop_type = pop_type:burghers }
		add = {
			value = location.pie_pop_local_burghers_wealth_scale
			subtract = 1
			min = 0
			multiply = 0.7
			max = 1.25
		}
	}
	else_if = {
		limit = {
			OR = {
				pop_type = pop_type:laborers
				pop_type = pop_type:peasants
				pop_type = pop_type:soldiers
			}
		}
		add = {
			value = location.pie_pop_local_commoner_wealth_scale
			subtract = 1
			min = 0
			multiply = 0.95
			max = 1.5
		}
	}
	else_if = {
		limit = { pop_type = pop_type:tribesmen }
		add = {
			value = location.pie_pop_local_tribesmen_wealth_scale
			subtract = 1
			min = 0
			multiply = 0.5
			max = 0.75
		}
	}
	else_if = {
		limit = { pop_type = pop_type:slaves }
		add = {
			value = location.pie_pop_local_commoner_wealth_scale
			subtract = 1
			min = 0
			multiply = 0.1
			max = 0.15
		}
	}

	max = 1.5
}
