pie_ab_remaining_build_capacity = {
	value = building_max_level
	subtract = building_level
	subtract = building_levels_under_construction
}

pie_ab_min_gold_with_spent = {
	value = var:pie_ab_best_cost
	if = {
		limit = { owner = { has_variable = pie_ab_gold_limit } }
		add = owner.var:pie_ab_gold_limit
	}
	if = {
		limit = { owner = { has_variable = pie_ab_spent } }
		add = owner.var:pie_ab_spent
	}
}

pie_ab_auto_build_enabled_count = {
	value = 0
	every_owned_location = {
		limit = { has_variable = pie_ab_auto_build }
		add = 1
	}
}

pie_ab_auto_build_non_core_enabled_count = {
	value = 0
	every_owned_location = {
		limit = {
			has_variable = pie_ab_auto_build
			NOT = { integration_level = core }
		}
		add = 1
	}
}
