pie_clear_land_owner = {
	remove_variable = pie_land_owner_crown
	remove_variable = pie_land_owner_nobles
	remove_variable = pie_land_owner_clergy
	remove_variable = pie_land_owner_burghers
	remove_variable = pie_land_owner_peasants

	remove_location_modifier = pie_land_owner_crown_modifier
	remove_location_modifier = pie_land_owner_nobles_modifier
	remove_location_modifier = pie_land_owner_clergy_modifier
	remove_location_modifier = pie_land_owner_burghers_modifier
	remove_location_modifier = pie_land_owner_peasants_modifier
}

pie_set_land_owner_crown = {
	pie_clear_land_owner = yes
	set_variable = { name = pie_land_owner_crown value = yes }
	add_location_modifier = { modifier = pie_land_owner_crown_modifier days = -1 mode = replace }
}

pie_grant_land_to_selected_estate = {
	if = {
		limit = { has_variable = pie_et_estate_nobles }
		ordered_owned_location = {
			limit = { pie_land_nobles_grant_candidate = yes }
			order_by = pie_land_grant_score
			max = 1
			pie_set_land_owner_nobles = yes
		}
	}
	else_if = {
		limit = { has_variable = pie_et_estate_clergy }
		ordered_owned_location = {
			limit = { pie_land_clergy_grant_candidate = yes }
			order_by = pie_land_grant_score
			max = 1
			pie_set_land_owner_clergy = yes
		}
	}
	else_if = {
		limit = { has_variable = pie_et_estate_burghers }
		ordered_owned_location = {
			limit = { pie_land_burghers_grant_candidate = yes }
			order_by = pie_land_grant_score
			max = 1
			pie_set_land_owner_burghers = yes
		}
	}
}

pie_apply_land_survey_estate_backlash = {
	if = {
		limit = { has_variable = pie_land_owner_nobles }
		owner = { add_estate_satisfaction = { type = estate_type:nobles_estate value = -0.06 } }
	}
	else_if = {
		limit = { has_variable = pie_land_owner_clergy }
		owner = { add_estate_satisfaction = { type = estate_type:clergy_estate value = -0.06 } }
	}
	else_if = {
		limit = { has_variable = pie_land_owner_burghers }
		owner = { add_estate_satisfaction = { type = estate_type:burghers_estate value = -0.06 } }
	}
	else_if = {
		limit = { has_variable = pie_land_owner_peasants }
		owner = { add_estate_satisfaction = { type = estate_type:peasants_estate value = -0.04 } }
	}
	else = {
		if = {
			limit = {
				local_relative_estate_power = {
					estate_type = estate_type:nobles_estate
					value >= 0.4
				}
			}
			owner = { add_estate_satisfaction = { type = estate_type:nobles_estate value = -0.02 } }
		}
		if = {
			limit = {
				local_relative_estate_power = {
					estate_type = estate_type:clergy_estate
					value >= 0.4
				}
			}
			owner = { add_estate_satisfaction = { type = estate_type:clergy_estate value = -0.02 } }
		}
		if = {
			limit = {
				local_relative_estate_power = {
					estate_type = estate_type:burghers_estate
					value >= 0.4
				}
			}
			owner = { add_estate_satisfaction = { type = estate_type:burghers_estate value = -0.02 } }
		}
		if = {
			limit = {
				local_relative_estate_power = {
					estate_type = estate_type:peasants_estate
					value >= 0.45
				}
			}
			owner = { add_estate_satisfaction = { type = estate_type:peasants_estate value = -0.01 } }
		}
	}
}

pie_set_passive_land_owner = {
	if = {
		limit = { pie_land_nobles_passive_claim_candidate = yes }

	else_if = {

	else_if = {

		pie_set_land_owner_burghers = yes
	}
	else_if = {
		limit = { pie_land_peasant_freehold_candidate = yes }
		pie_set_land_owner_peasants = yes
	}
}

pie_progress_natural_land_integration = {
	change_integration_progress = pie_land_natural_integration_progress
	if = {
		limit = { integration_progress >= 100 }
		change_integration_level = integrated
	}
	if = {
		limit = {
			OR = {
				integration_level = integrated
				integration_level = core
			}
			NOT = { pie_land_has_defined_owner = yes }
		}
		pie_set_natural_integrated_land_owner = yes
	}
}

pie_complete_crown_land_survey = {
	if = {
		limit = { pie_bureaucracy_tier_4 = yes }
		scope:target = {
			ordered_location_in_province = {
				limit = {
					owner ?= scope:actor
					controller ?= scope:actor
					pie_land_crown_survey_candidate = yes
				}
				order_by = pie_land_survey_score
				max = 5
				pie_apply_land_survey_estate_backlash = yes
				pie_set_land_owner_crown = yes
			}
		}
	}
	else = {
		scope:target = {
			ordered_location_in_province = {
				limit = {
					owner ?= scope:actor
					controller ?= scope:actor
					pie_land_crown_survey_candidate = yes
				}
				order_by = pie_land_survey_score
				max = 3
				pie_apply_land_survey_estate_backlash = yes
				pie_set_land_owner_crown = yes
			}
		}
	}
}

pie_land_ownership_yearly = {
	if = {
		limit = {
			any_owned_location = { pie_land_natural_integration_candidate = yes }
		}
		ordered_owned_location = {
			limit = { pie_land_natural_integration_candidate = yes }
			order_by = pie_land_natural_integration_score
			max = 10
			pie_progress_natural_land_integration = yes
		}
	}

	if = {
		limit = {
			any_owned_location = { pie_land_passive_claim_candidate = yes }
		}
		ordered_owned_location = {
			limit = { pie_land_passive_claim_candidate = yes }
			order_by = pie_land_passive_claim_score
			max = 1
			pie_set_passive_land_owner = yes
		}
	}
}
