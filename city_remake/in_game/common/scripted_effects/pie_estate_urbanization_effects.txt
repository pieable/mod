pie_clear_urban_origin = {
	if = { limit = { has_variable = pie_pending_origin_nobles_town } remove_variable = pie_pending_origin_nobles_town }
	if = { limit = { has_variable = pie_pending_origin_nobles_city } remove_variable = pie_pending_origin_nobles_city }
	if = { limit = { has_variable = pie_pending_origin_clergy_town } remove_variable = pie_pending_origin_clergy_town }
	if = { limit = { has_variable = pie_pending_origin_clergy_city } remove_variable = pie_pending_origin_clergy_city }
	if = { limit = { has_variable = pie_pending_origin_burghers_town } remove_variable = pie_pending_origin_burghers_town }
	if = { limit = { has_variable = pie_pending_origin_burghers_city } remove_variable = pie_pending_origin_burghers_city }

	if = { limit = { has_variable = pie_origin_nobles_town } remove_variable = pie_origin_nobles_town }
	if = { limit = { has_variable = pie_origin_nobles_city } remove_variable = pie_origin_nobles_city }
	if = { limit = { has_variable = pie_origin_clergy_town } remove_variable = pie_origin_clergy_town }
	if = { limit = { has_variable = pie_origin_clergy_city } remove_variable = pie_origin_clergy_city }
	if = { limit = { has_variable = pie_origin_burghers_town } remove_variable = pie_origin_burghers_town }
	if = { limit = { has_variable = pie_origin_burghers_city } remove_variable = pie_origin_burghers_city }
	if = { limit = { has_variable = pie_origin_player_town } remove_variable = pie_origin_player_town }
	if = { limit = { has_variable = pie_origin_player_city } remove_variable = pie_origin_player_city }
}

pie_destroy_estate_urban_building_only = {
	every_buildings_in_location = {
		limit = {
			OR = {
				building_type = building_type:pie_noble_manor_town
				building_type = building_type:pie_noble_frontier_town
				building_type = building_type:pie_noble_court_city
				building_type = building_type:pie_clergy_parish_town
				building_type = building_type:pie_clergy_scholastic_city
				building_type = building_type:pie_burghers_market_town
				building_type = building_type:pie_burghers_exchange_city
			}
		}
		location = { destroy_building = prev }
	}
	pie_clear_urban_origin = yes
}

pie_destroy_tracked_urban_location = {
	pie_destroy_estate_urban_building_only = yes
	change_location_rank = location_rank:rural_settlement
}

pie_downgrade_tracked_city_to_town = {
	every_buildings_in_location = {
		limit = {
			OR = {
				building_type = building_type:pie_noble_court_city
				building_type = building_type:pie_clergy_scholastic_city
				building_type = building_type:pie_burghers_exchange_city
			}
		}
		location = { destroy_building = prev }
	}

	if = {
		limit = { has_variable = pie_origin_nobles_city }
		remove_variable = pie_origin_nobles_city
		set_variable = { name = pie_origin_nobles_town value = yes }
	}
	if = {
		limit = { has_variable = pie_origin_clergy_city }
		remove_variable = pie_origin_clergy_city
		set_variable = { name = pie_origin_clergy_town value = yes }
	}
	if = {
		limit = { has_variable = pie_origin_burghers_city }
		remove_variable = pie_origin_burghers_city
		set_variable = { name = pie_origin_burghers_town value = yes }
	}
	if = {
		limit = { has_variable = pie_origin_player_city }
		remove_variable = pie_origin_player_city
		set_variable = { name = pie_origin_player_town value = yes }
	}

	change_location_rank = location_rank:town
}
