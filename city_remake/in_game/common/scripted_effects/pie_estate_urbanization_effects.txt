	if = { limit = { has_variable = pie_pending_origin_nobles_city } remove_variable = pie_pending_origin_nobles_city }
	if = { limit = { has_variable = pie_pending_origin_clergy_town } remove_variable = pie_pending_origin_clergy_town }
	if = { limit = { has_variable = pie_pending_origin_clergy_city } remove_variable = pie_pending_origin_clergy_city }
	if = { limit = { has_variable = pie_pending_origin_burghers_town } remove_variable = pie_pending_origin_burghers_town }
	if = { limit = { has_variable = pie_pending_origin_burghers_city } remove_variable = pie_pending_origin_burghers_city }

	if = { limit = { has_variable = pie_origin_nobles_town } remove_variable = pie_origin_nobles_town }
	if = { limit = { has_variable = pie_origin_nobles_city } remove_variable = pie_origin_nobles_city }
	if = { limit = { has_variable = pie_origin_clergy_town } remove_variable = pie_origin_clergy_town }
	if = { limit = { has_variable = pie_origin_clergy_city } remove_variable = pie_origin_clergy_city }
	if = { limit = { has_variable = pie_origin_burghers_town } remove_variable = pie_origin_burghers_town }
	if = { limit = { has_variable = pie_origin_burghers_city } remove_variable = pie_origin_burghers_city }
	if = { limit = { has_variable = pie_origin_player_town } remove_variable = pie_origin_player_town }
	if = { limit = { has_variable = pie_origin_player_city } remove_variable = pie_origin_player_city }
}

pie_destroy_estate_urban_building_only = {
	every_buildings_in_location = {
		limit = {
			OR = {
				building_type = building_type:pie_noble_manor_town
				building_type = building_type:pie_noble_frontier_town
				building_type = building_type:pie_noble_court_city
				building_type = building_type:pie_clergy_parish_town
				building_type = building_type:pie_clergy_scholastic_city
				building_type = building_type:pie_burghers_market_town
				building_type = building_type:pie_burghers_exchange_city
			}
		}
		location = { destroy_building = prev }
	}
	pie_clear_urban_origin = yes
}

pie_destroy_tracked_urban_location = {
	pie_destroy_estate_urban_building_only = yes
	change_location_rank = location_rank:rural_settlement
}

pie_downgrade_tracked_city_to_town = {
	every_buildings_in_location = {
		limit = {
			OR = {
				building_type = building_type:pie_noble_court_city
				building_type = building_type:pie_clergy_scholastic_city
				building_type = building_type:pie_burghers_exchange_city
			}
		}
		location = { destroy_building = prev }
	}

	if = {
		limit = { has_variable = pie_origin_nobles_city }
		remove_variable = pie_origin_nobles_city
		set_variable = { name = pie_origin_nobles_town value = yes }
	}
	if = {
		limit = { has_variable = pie_origin_clergy_city }
		remove_variable = pie_origin_clergy_city
		set_variable = { name = pie_origin_clergy_town value = yes }
	}
	if = {
		limit = { has_variable = pie_origin_burghers_city }
		remove_variable = pie_origin_burghers_city
		set_variable = { name = pie_origin_burghers_town value = yes }
	}
	if = {
		limit = { has_variable = pie_origin_player_city }
		remove_variable = pie_origin_player_city
		set_variable = { name = pie_origin_player_town value = yes }
	}

	change_location_rank = location_rank:town
}

pie_clear_urbanization_project = {
	if = { limit = { has_variable = pie_town_project_active } remove_variable = pie_town_project_active }
	if = { limit = { has_variable = pie_city_project_active } remove_variable = pie_city_project_active }
	if = { limit = { has_variable = pie_project_estate_nobles } remove_variable = pie_project_estate_nobles }
	if = { limit = { has_variable = pie_project_estate_clergy } remove_variable = pie_project_estate_clergy }
	if = { limit = { has_variable = pie_project_estate_burghers } remove_variable = pie_project_estate_burghers }
}

pie_assign_urbanization_project_estate = {
	if = { limit = { has_variable = pie_project_estate_nobles } remove_variable = pie_project_estate_nobles }
	if = { limit = { has_variable = pie_project_estate_clergy } remove_variable = pie_project_estate_clergy }
	if = { limit = { has_variable = pie_project_estate_burghers } remove_variable = pie_project_estate_burghers }

	if = {
		limit = {
			OR = {
				has_variable = pie_origin_nobles_town
				has_variable = pie_origin_nobles_city
			}
			owner = { country_has_estate = estate_type:nobles_estate }
		}
		set_variable = { name = pie_project_estate_nobles value = yes }
	}
	else_if = {
		limit = {
			OR = {
				has_variable = pie_origin_clergy_town
				has_variable = pie_origin_clergy_city
			}
			owner = { country_has_estate = estate_type:clergy_estate }
		}
		set_variable = { name = pie_project_estate_clergy value = yes }
	}
	else_if = {
		limit = {
			OR = {
				has_variable = pie_origin_burghers_town
				has_variable = pie_origin_burghers_city
			}
			owner = { country_has_estate = estate_type:burghers_estate }
		}
		set_variable = { name = pie_project_estate_burghers value = yes }
	}
	else_if = {
		limit = {
			owner = {
				country_has_estate = estate_type:nobles_estate
				country_has_estate = estate_type:clergy_estate
				country_has_estate = estate_type:burghers_estate
			}
		}
		random_list = {
			1 = { set_variable = { name = pie_project_estate_nobles value = yes } }
			1 = { set_variable = { name = pie_project_estate_clergy value = yes } }
			1 = { set_variable = { name = pie_project_estate_burghers value = yes } }
		}
	}
	else_if = {
		limit = {
			owner = {
				country_has_estate = estate_type:nobles_estate
				country_has_estate = estate_type:clergy_estate
			}
		}
		random_list = {
			1 = { set_variable = { name = pie_project_estate_nobles value = yes } }
			1 = { set_variable = { name = pie_project_estate_clergy value = yes } }
		}
	}
	else_if = {
		limit = {
			owner = {
				country_has_estate = estate_type:nobles_estate
				country_has_estate = estate_type:burghers_estate
			}
		}
		random_list = {
			1 = { set_variable = { name = pie_project_estate_nobles value = yes } }
			1 = { set_variable = { name = pie_project_estate_burghers value = yes } }
		}
	}
	else_if = {
		limit = {
			owner = {
				country_has_estate = estate_type:clergy_estate
				country_has_estate = estate_type:burghers_estate
			}
		}
		random_list = {
			1 = { set_variable = { name = pie_project_estate_clergy value = yes } }
			1 = { set_variable = { name = pie_project_estate_burghers value = yes } }
		}
	}
	else_if = {
		limit = { owner = { country_has_estate = estate_type:nobles_estate } }
		set_variable = { name = pie_project_estate_nobles value = yes }
	}
	else_if = {
		limit = { owner = { country_has_estate = estate_type:clergy_estate } }
		set_variable = { name = pie_project_estate_clergy value = yes }
	}
	else_if = {
		limit = { owner = { country_has_estate = estate_type:burghers_estate } }
		set_variable = { name = pie_project_estate_burghers value = yes }
	}
}

		set_variable = { name = pie_pending_origin_nobles_town value = yes }

		construct_estate_building = {
			building_type = building_type:pie_noble_manor_town
			estate_type = estate_type:nobles_estate
		}
		remove_variable = pie_pending_origin_nobles_town

		set_variable = { name = pie_pending_origin_clergy_town value = yes }

			building_type = building_type:pie_clergy_parish_town
			estate_type = estate_type:clergy_estate
		}
		remove_variable = pie_pending_origin_clergy_town
		set_variable = { name = pie_origin_clergy_town value = yes }
	}
	else_if = {
		limit = { has_variable = pie_project_estate_burghers }
		set_variable = { name = pie_pending_origin_burghers_town value = yes }
		pie_clear_urban_origin = yes
		change_location_rank = location_rank:town

pie_advance_town_urbanization_project = {
	if = {
		limit = { pie_town_project_can_complete = yes }
		if = {
			limit = { owner = { is_ai = no } }
			pie_trigger_town_urbanization_project_event = yes
		}
		else = {
			pie_complete_town_urbanization_project = yes
		}
	}
	else_if = {
		limit = {
			location_rank = location_rank:rural_settlement
			num_civil_constructions < 1
			pie_project_estate_can_pay_rural_infrastructure = yes
		}
		pie_build_next_town_project_infrastructure = yes
	}

pie_advance_noble_urbanization_projects = {
	if = {
		limit = {
			var:pie_project_budget_nobles > 0
			pie_nobles_can_fund_reserved_urbanization_budget = yes
		}
		every_owned_location = {
			limit = {
				has_variable = pie_town_project_active
				has_variable = pie_project_estate_nobles
				NOT = { pie_town_project_can_complete = yes }
			}
			pie_advance_town_urbanization_project = yes
		}
		every_owned_location = {
			limit = {
				has_variable = pie_city_project_active
				has_variable = pie_project_estate_nobles
				NOT = { pie_city_project_can_complete = yes }
			}
			pie_advance_city_urbanization_project = yes
		}
	}
	else_if = {
		limit = {
			OR = {
				AND = {
					"estate(estate_type:nobles_estate)" = { estate_gold >= 200 }
					any_owned_location = {
						has_variable = pie_city_project_active
						has_variable = pie_project_estate_nobles
						NOT = { pie_city_project_can_complete = yes }
					}
				}
				AND = {
					"estate(estate_type:nobles_estate)" = { estate_gold >= 100 }
					any_owned_location = {
						has_variable = pie_town_project_active
						has_variable = pie_project_estate_nobles
						NOT = { pie_town_project_can_complete = yes }
					}
				}
			}
		}
		random_list = {
			1 = {
				trigger = {
					"estate(estate_type:nobles_estate)" = { estate_gold >= 200 }
					any_owned_location = {
						has_variable = pie_city_project_active
						has_variable = pie_project_estate_nobles
						NOT = { pie_city_project_can_complete = yes }
					}
				}
				set_variable = { name = pie_project_budget_nobles value = 200 }
				random_owned_location = {
					limit = {
						has_variable = pie_city_project_active
						has_variable = pie_project_estate_nobles
						NOT = { pie_city_project_can_complete = yes }
					}
					pie_advance_city_urbanization_project = yes
				}
			}
			1 = {
				trigger = {
					"estate(estate_type:nobles_estate)" = { estate_gold >= 100 }
					any_owned_location = {
						has_variable = pie_town_project_active
						has_variable = pie_project_estate_nobles
						NOT = { pie_town_project_can_complete = yes }
					}
				}
				set_variable = { name = pie_project_budget_nobles value = 100 }
				random_owned_location = {
					limit = {
						has_variable = pie_town_project_active
						has_variable = pie_project_estate_nobles
						NOT = { pie_town_project_can_complete = yes }
					}
					pie_advance_town_urbanization_project = yes
				}
			}
		}
	}
}

		limit = {
			has_variable = pie_town_project_active
			NOT = { pie_town_project_still_viable = yes }
		}
		pie_clear_urbanization_project = yes
	}
	if = {
		limit = {
			has_variable = pie_city_project_active
			NOT = { pie_city_project_still_viable = yes }
		}
		pie_clear_urbanization_project = yes
	}
	if = {
		limit = {
			pie_has_active_urbanization_project = yes
			NOT = { pie_has_project_estate = yes }
		}
		pie_clear_urbanization_project = yes
	}
}

pie_estate_urbanization_projects_yearly = {
	every_owned_location = {
		pie_clean_invalid_urbanization_project = yes
	}

	every_owned_location = {
		limit = {
			has_variable = pie_town_project_active
			pie_town_project_can_complete = yes
		}
		pie_advance_town_urbanization_project = yes
	}

	every_owned_location = {
		limit = {
			has_variable = pie_city_project_active
			pie_city_project_can_complete = yes
		}
		pie_advance_city_urbanization_project = yes
	}

	pie_prepare_urbanization_project_budgets = yes
	pie_advance_noble_urbanization_projects = yes
	pie_advance_clergy_urbanization_projects = yes
	pie_advance_burghers_urbanization_projects = yes

	if = {
		limit = { pie_can_start_new_urbanization_project = yes }
		set_variable = { name = pie_town_project_count value = 0 }
		set_variable = { name = pie_city_project_count value = 0 }

		every_owned_location = {
			limit = { has_variable = pie_town_project_active }
			owner = { change_variable = { name = pie_town_project_count add = 1 } }
		}
		every_owned_location = {
			limit = { has_variable = pie_city_project_active }
			owner = { change_variable = { name = pie_city_project_count add = 1 } }
		}
		pie_prepare_urbanization_project_caps = yes

		if = {
			limit = {
				var:pie_town_project_count < var:pie_town_project_cap
				any_owned_location = { pie_noble_town_project_candidate = yes }
			}
			ordered_owned_location = {
				limit = { pie_noble_town_project_candidate = yes }
				order_by = pie_estate_urbanization_project_score
				max = 1
				pie_create_noble_town_urbanization_project = yes
				owner = { change_variable = { name = pie_town_project_count add = 1 } }
			}
		}

		if = {
			limit = {
				var:pie_town_project_count < var:pie_town_project_cap
				any_owned_location = { pie_clergy_town_project_candidate = yes }
			}
			ordered_owned_location = {
				limit = { pie_clergy_town_project_candidate = yes }
				order_by = pie_estate_urbanization_project_score
				max = 1
				pie_create_clergy_town_urbanization_project = yes
				owner = { change_variable = { name = pie_town_project_count add = 1 } }
			}
		}

		if = {
			limit = {
				var:pie_town_project_count < var:pie_town_project_cap
				any_owned_location = { pie_burghers_town_project_candidate = yes }
			}
			ordered_owned_location = {
				limit = { pie_burghers_town_project_candidate = yes }
				order_by = pie_estate_urbanization_project_score
				max = 1
				pie_create_burghers_town_urbanization_project = yes
				owner = { change_variable = { name = pie_town_project_count add = 1 } }
			}
		}

		if = {
			limit = {
				var:pie_city_project_count < var:pie_city_project_cap
				any_owned_location = { pie_noble_city_project_candidate = yes }
			}
			ordered_owned_location = {
				limit = { pie_noble_city_project_candidate = yes }
				order_by = pie_estate_urbanization_project_score
				max = 1
				pie_create_noble_city_urbanization_project = yes
				owner = { change_variable = { name = pie_city_project_count add = 1 } }
			}
		}

		if = {
			limit = {
				var:pie_city_project_count < var:pie_city_project_cap
				any_owned_location = { pie_clergy_city_project_candidate = yes }
			}
			ordered_owned_location = {
				limit = { pie_clergy_city_project_candidate = yes }
				order_by = pie_estate_urbanization_project_score
				max = 1
				pie_create_clergy_city_urbanization_project = yes
				owner = { change_variable = { name = pie_city_project_count add = 1 } }
			}
		}

		if = {
			limit = {
				var:pie_city_project_count < var:pie_city_project_cap
				any_owned_location = { pie_burghers_city_project_candidate = yes }
			}
			ordered_owned_location = {
				limit = { pie_burghers_city_project_candidate = yes }
				order_by = pie_estate_urbanization_project_score
				max = 1
				pie_create_burghers_city_urbanization_project = yes
				owner = { change_variable = { name = pie_city_project_count add = 1 } }
			}
		}

		remove_variable = pie_town_project_count
		remove_variable = pie_city_project_count
		pie_clear_urbanization_project_caps = yes
	}

	pie_clear_urbanization_project_budgets = yes
}
