noicthy_apply_occupation = {
	set_local_variable = { name = noicthy_occupied_forts value = 0 }
	set_local_variable = { name = noicthy_occupied_non_forts value = 0 }
	set_local_variable = { name = noicthy_total_forts value = num_forts }
	set_local_variable = { name = noicthy_total_non_forts value = { add = num_locations subtract = num_forts } }

	every_owned_location = {
		limit = { NOT = { controller = prev } }
		if = {
			limit = { has_fort = yes }
			change_local_variable = { name = noicthy_occupied_forts add = 1 }
		}
		else = {
			change_local_variable = { name = noicthy_occupied_non_forts add = 1 }
		}
	}

	if = {
		limit = { has_variable = noicthy_was_auto_conquered }
		save_temporary_scope_as = noicthy_ac_victim

		every_war = {
			limit = { is_in_war = scope:noicthy_ac_victim }
			every_war_participant = {
				limit = { NOT = { is_fighting_war_together_with = scope:noicthy_ac_victim } }
				every_owned_location = {
					limit = {
						has_variable = noicthy_original_owner
						exists = var:noicthy_original_owner
						var:noicthy_original_owner = scope:noicthy_ac_victim
					}

					if = {
						limit = { has_fort = yes }
						change_local_variable = { name = noicthy_total_forts add = 1 }
					}
					else = {
						change_local_variable = { name = noicthy_total_non_forts add = 1 }
					}

					if = {
						limit = { NOT = { controller = scope:noicthy_ac_victim } }
						if = {
							limit = { has_fort = yes }
							change_local_variable = { name = noicthy_occupied_forts add = 1 }
						}
						else = {
							change_local_variable = { name = noicthy_occupied_non_forts add = 1 }
						}
					}
				}
			}
		}
	}

	set_local_variable = {
		name = noicthy_total_location_weight
		value = {
			add = local_var:noicthy_total_forts
			multiply = 5
			add = local_var:noicthy_total_non_forts
		}
	}
	set_local_variable = {
		name = noicthy_occupied_location_weight
		value = {
			add = local_var:noicthy_occupied_forts
			multiply = 5
			add = local_var:noicthy_occupied_non_forts
		}
	}
	set_local_variable = { name = noicthy_occupation_size value = 0 }

	if = {
		limit = { local_var:noicthy_total_location_weight > 0 }
		set_local_variable = {
			name = noicthy_occupation_size
			value = {
				add = local_var:noicthy_occupied_location_weight
				divide = local_var:noicthy_total_location_weight
			}
		}
	}
	if = {
		limit = { local_var:noicthy_occupation_size > 1 }
		set_local_variable = { name = noicthy_occupation_size value = 1 }
	}
}

noicthy_update_territory = {
	noicthy_apply_occupation = yes

	set_local_variable = { name = noicthy_civil_war_penalty value = 0 }
	set_local_variable = { name = noicthy_civil_war_my_weight value = 0 }
	set_local_variable = { name = noicthy_civil_war_opponent_weight value = 0 }

	if = {
		limit = { in_civil_war = yes }
		set_local_variable = {
			name = noicthy_civil_war_my_weight
			value = { add = num_forts multiply = 4 add = num_locations }
		}
		civil_war_opponent = {
			set_local_variable = {
				name = noicthy_civil_war_opponent_weight
				value = { add = num_forts multiply = 4 add = num_locations }
			}
		}

		if = {
			limit = {
				local_var:noicthy_civil_war_opponent_weight > 0
				local_var:noicthy_civil_war_my_weight < local_var:noicthy_civil_war_opponent_weight
			}
			set_local_variable = {
				name = noicthy_civil_war_penalty
				value = {
					add = 1
					subtract = {
						add = local_var:noicthy_civil_war_my_weight
						divide = local_var:noicthy_civil_war_opponent_weight
					}
				}
			}
		}
	}

	set_local_variable = { name = noicthy_effective_occupation value = local_var:noicthy_occupation_size }
	if = {
		limit = { local_var:noicthy_civil_war_penalty > local_var:noicthy_effective_occupation }
		set_local_variable = { name = noicthy_effective_occupation value = local_var:noicthy_civil_war_penalty }
	}
	if = {
		limit = { local_var:noicthy_effective_occupation > 1 }
		set_local_variable = { name = noicthy_effective_occupation value = 1 }
	}

	remove_country_modifier = noicthy_war_momentum_occupation
	if = {
		limit = { local_var:noicthy_effective_occupation > 0 }
		add_country_modifier = {
			modifier = noicthy_war_momentum_occupation
			days = -1
			size = local_var:noicthy_effective_occupation
		}
	}

	set_local_variable = {
		name = noicthy_combined_size
		value = {
			add = var:noicthy_war_momentum_army_size
			add = local_var:noicthy_effective_occupation
		}
	}

	capital ?= {
		remove_location_modifier = noicthy_war_momentum_capital
		if = {
			limit = { local_var:noicthy_combined_size > 0 }
			add_location_modifier = {
				modifier = noicthy_war_momentum_capital
				days = -1
				size = local_var:noicthy_combined_size
			}
		}
	}

	remove_country_modifier = noicthy_war_momentum_capital_fallen
	if = {
		limit = {
			capital ?= { NOT = { controller = prev } }
		}
		add_country_modifier = {
			modifier = noicthy_war_momentum_capital_fallen
			days = -1
		}
	}
}
