pie_et_rebuild_land_lists = {
	clear_variable_list = pie_et_sell_land_locations
	clear_variable_list = pie_et_buy_land_locations

	every_owned_location = {
		limit = { pie_et_sell_land_candidate = yes }
		ROOT = {
			add_to_variable_list = {
				name = pie_et_sell_land_locations
				target = prev
			}
		}
	}

	every_owned_location = {
		limit = { pie_et_buy_land_candidate = yes }
		ROOT = {
			add_to_variable_list = {
				name = pie_et_buy_land_locations
				target = prev
			}
		}
	}
}

pie_et_init_trade = {
	if = {
		limit = {
			NOT = { has_variable = pie_et_estate_nobles }
			NOT = { has_variable = pie_et_estate_clergy }
			NOT = { has_variable = pie_et_estate_burghers }
		}
		if = {
			limit = { country_has_estate = estate_type:nobles_estate }
			set_variable = { name = pie_et_estate_nobles value = yes }
		}
		else_if = {
			limit = { country_has_estate = estate_type:clergy_estate }
			set_variable = { name = pie_et_estate_clergy value = yes }
		}
		else_if = {
			limit = { country_has_estate = estate_type:burghers_estate }
			set_variable = { name = pie_et_estate_burghers value = yes }
		}
	}
	if = {
		limit = {
			NOT = { has_variable = pie_et_category_land }
			NOT = { has_variable = pie_et_category_privilege }
			NOT = { has_variable = pie_et_category_other }
		}
		set_variable = { name = pie_et_category_land value = yes }
	}
	if = {
		limit = { NOT = { has_variable = pie_et_gold } }
		set_variable = { name = pie_et_gold value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = pie_et_interest } }
		set_variable = { name = pie_et_interest value = 0 }
	}
	pie_et_rebuild_land_lists = yes
}

pie_et_apply_selected_estate_gold = {
	if = {
		limit = { var:pie_et_gold > 0 }
		add_gold = { value = var:pie_et_gold }
		if = {
			limit = { has_variable = pie_et_estate_nobles }
			add_gold_to_estate = {
				estate_type = estate_type:nobles_estate
				value = { value = var:pie_et_gold multiply = -1 }
			}
			change_variable = { name = pie_et_debt_nobles add = { value = var:pie_et_gold } }
			set_variable = { name = pie_et_interest_nobles value = var:pie_et_interest }
		}
		else_if = {
			limit = { has_variable = pie_et_estate_clergy }
			add_gold_to_estate = {
				estate_type = estate_type:clergy_estate
				value = { value = var:pie_et_gold multiply = -1 }
			}
			change_variable = { name = pie_et_debt_clergy add = { value = var:pie_et_gold } }
			set_variable = { name = pie_et_interest_clergy value = var:pie_et_interest }
		}
		else_if = {
			limit = { has_variable = pie_et_estate_burghers }
			add_gold_to_estate = {
				estate_type = estate_type:burghers_estate
				value = { value = var:pie_et_gold multiply = -1 }
			}
			change_variable = { name = pie_et_debt_burghers add = { value = var:pie_et_gold } }
			set_variable = { name = pie_et_interest_burghers value = var:pie_et_interest }
		}
	}
	else_if = {
		limit = { var:pie_et_gold < 0 }
		add_gold = { value = var:pie_et_gold }
		if = {
			limit = { has_variable = pie_et_estate_nobles }
			add_gold_to_estate = {
				estate_type = estate_type:nobles_estate
				value = { value = var:pie_et_gold multiply = -1 }
			}
		}
		else_if = {
			limit = { has_variable = pie_et_estate_clergy }
			add_gold_to_estate = {
				estate_type = estate_type:clergy_estate
				value = { value = var:pie_et_gold multiply = -1 }
			}
		}
		else_if = {
			limit = { has_variable = pie_et_estate_burghers }
			add_gold_to_estate = {
				estate_type = estate_type:burghers_estate
				value = { value = var:pie_et_gold multiply = -1 }
			}
		}
	}
}

pie_et_apply_selected_land_trades = {
	every_owned_location = {
		limit = {
			has_variable = pie_et_selected_sell_land
			pie_et_sell_land_candidate = yes
		}
		if = {
			limit = { owner = { has_variable = pie_et_estate_nobles } }
			pie_set_land_owner_nobles = yes
			owner = {
				add_estate_satisfaction = { type = estate_type:nobles_estate value = 0.03 }
			}
		}
		else_if = {
			limit = { owner = { has_variable = pie_et_estate_clergy } }
			pie_set_land_owner_clergy = yes
			owner = {
				add_estate_satisfaction = { type = estate_type:clergy_estate value = 0.03 }
			}
		}
		else_if = {
			limit = { owner = { has_variable = pie_et_estate_burghers } }
			pie_set_land_owner_burghers = yes
			owner = {
				add_estate_satisfaction = { type = estate_type:burghers_estate value = 0.03 }
			}
		}
		remove_variable = pie_et_selected_sell_land
	}

	every_owned_location = {
		limit = {
			has_variable = pie_et_selected_buy_land
			pie_et_buy_land_candidate = yes
		}
		if = {
			limit = { owner = { has_variable = pie_et_estate_nobles } }
			owner = {
				add_estate_satisfaction = { type = estate_type:nobles_estate value = -0.05 }
			}
		}
		else_if = {
			limit = { owner = { has_variable = pie_et_estate_clergy } }
			owner = {
				add_estate_satisfaction = { type = estate_type:clergy_estate value = -0.05 }
			}
		}
		else_if = {
			limit = { owner = { has_variable = pie_et_estate_burghers } }
			owner = {
				add_estate_satisfaction = { type = estate_type:burghers_estate value = -0.05 }
			}
		}
		pie_set_land_owner_crown = yes
		remove_variable = pie_et_selected_buy_land
	}

	pie_et_rebuild_land_lists = yes
}
