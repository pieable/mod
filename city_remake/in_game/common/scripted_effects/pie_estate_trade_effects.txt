pie_et_toggle_estate_picker = {
	if = {
		limit = { has_variable = pie_et_estate_picker_open }
		remove_variable = pie_et_estate_picker_open
	}
	else = {
		set_variable = { name = pie_et_estate_picker_open value = yes }
	}
}

pie_et_rebuild_land_lists = {
	save_temporary_scope_as = pie_et_land_list_owner
	clear_variable_list = pie_et_sell_land_locations
	clear_variable_list = pie_et_grant_unregistered_land_locations
	clear_variable_list = pie_et_buy_land_locations
	clear_variable_list = pie_et_selected_sell_land_locations
	clear_variable_list = pie_et_selected_grant_unregistered_land_locations
	clear_variable_list = pie_et_selected_buy_land_locations

	every_owned_location = {
		limit = {
			has_variable = pie_et_selected_sell_land
			pie_et_sell_land_candidate = yes
		}
		scope:pie_et_land_list_owner = {
			add_to_variable_list = {
				name = pie_et_selected_sell_land_locations
				target = prev
			}
		}
	}

	every_owned_location = {
		limit = {
			has_variable = pie_et_selected_grant_unregistered_land
			pie_et_grant_unregistered_land_candidate = yes
		}
		scope:pie_et_land_list_owner = {
			add_to_variable_list = {
				name = pie_et_selected_grant_unregistered_land_locations
				target = prev
			}
		}
	}

	every_owned_location = {
		limit = {
			has_variable = pie_et_selected_buy_land

		}
		scope:pie_et_land_list_owner = {
			add_to_variable_list = {
				name = pie_et_selected_buy_land_locations
				target = prev
			}
		}
	}
}

			}
		}
	}
}

pie_et_init_trade = {
	if = {
		limit = {
			NOT = { has_variable = pie_et_estate_nobles }
			NOT = { has_variable = pie_et_estate_clergy }
			NOT = { has_variable = pie_et_estate_burghers }
		}
		if = {
			limit = { country_has_estate = estate_type:nobles_estate }
			set_variable = { name = pie_et_estate_nobles value = yes }
		}
		else_if = {
			limit = { country_has_estate = estate_type:clergy_estate }
			set_variable = { name = pie_et_estate_clergy value = yes }
		}
		else_if = {
			limit = { country_has_estate = estate_type:burghers_estate }
			set_variable = { name = pie_et_estate_burghers value = yes }
		}
	}
	if = {
		limit = {
			NOT = { has_variable = pie_et_category_land }
			NOT = { has_variable = pie_et_category_project }
			NOT = { has_variable = pie_et_category_privilege }
			NOT = { has_variable = pie_et_category_other }
		}
		set_variable = { name = pie_et_category_land value = yes }
	}
	if = {
		limit = { NOT = { has_variable = pie_et_gold } }
		set_variable = { name = pie_et_gold value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = pie_et_interest } }
		set_variable = { name = pie_et_interest value = 0 }
	}
	if = {
		limit = { var:pie_et_gold <= 0 }
		set_variable = { name = pie_et_interest value = 0 }
	}
	if = {
		limit = { var:pie_et_interest > 0.25 }
		set_variable = { name = pie_et_interest value = 0.25 }
	}
	pie_et_migrate_legacy_debts = yes
	pie_et_rebuild_land_lists = yes
	pie_et_rebuild_project_lists = yes
}

		}

		set_variable = { name = pie_et_debt_due_pending value = yes }

		set_variable = { name = $slot_marker$ value = yes }

	}
}

	}
}

pie_et_pay_debt_slot_monthly_interest = {
	if = {
		limit = {
			has_variable = $debt_var$
			var:$debt_var$ > 0
			NOT = { has_variable = $months_var$ }
		}
		set_variable = { name = $months_var$ value = 60 }
	}
	if = {
		limit = {
			has_variable = $debt_var$
			var:$debt_var$ > 0
			has_variable = $months_var$
			var:$months_var$ > 0
		}
		if = {
			limit = {
				has_variable = $interest_var$
				var:$interest_var$ > 0
			}
			if = {
				limit = {
					gold >= { value = var:$debt_var$ multiply = var:$interest_var$ divide = 12 }
				}
				add_gold = { value = var:$debt_var$ multiply = var:$interest_var$ divide = -12 }
				add_gold_to_estate = {
					estate_type = estate_type:$estate$
					value = { value = var:$debt_var$ multiply = var:$interest_var$ divide = 12 }
				}
			}
			else = {
				add_estate_satisfaction = { type = estate_type:$estate$ value = -0.01 }
			}
		}
		change_variable = { name = $months_var$ add = -1 }
	}
	if = {
		limit = {
			has_variable = $debt_var$
			var:$debt_var$ > 0
			has_variable = $months_var$
			var:$months_var$ <= 0
			NOT = { has_variable = pie_et_debt_due_pending }
		}
		pie_et_queue_due_debt_slot = {
			debt_var = $debt_var$
			interest_var = $interest_var$
			estate_marker = $estate_marker$
			slot_marker = $slot_marker$
		}
	}
}
