monthly_country_pulse = {
	on_actions = {
		pie_siege_starvation_monthly
	}
}

pie_siege_starvation_monthly = {
	effect = {
		every_owned_location = {
			if = {
				limit = { has_siege = yes }
				pie_apply_starving_siege_pressure = yes
			}
			else_if = {
				limit = { has_variable = pie_starving_siege_months }
				pie_apply_starving_siege_pressure = yes
			}
		}
	}
}
