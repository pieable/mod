pie_population_downgrade_monthly = {
	effect = {
		every_owned_location = {
			if = {
				limit = { pie_city_should_auto_downgrade = yes }
				if = {
					limit = { has_variable = pie_city_underpopulated_months }
					change_variable = { name = pie_city_underpopulated_months add = 1 }
				}
				else = {
					set_variable = { name = pie_city_underpopulated_months value = 1 }
				}
				if = {
					limit = { var:pie_city_underpopulated_months >= 12 }
					pie_downgrade_tracked_city_to_town = yes
					if = { limit = { has_variable = pie_city_underpopulated_months } remove_variable = pie_city_underpopulated_months }
					if = { limit = { has_variable = pie_town_underpopulated_months } remove_variable = pie_town_underpopulated_months }
				}
			}
			else_if = {
				limit = { pie_town_should_auto_downgrade = yes }
				if = {
					limit = { has_variable = pie_town_underpopulated_months }
					change_variable = { name = pie_town_underpopulated_months add = 1 }
				}
				else = {
					set_variable = { name = pie_town_underpopulated_months value = 1 }
				}
				if = {
					limit = { var:pie_town_underpopulated_months >= 12 }
					pie_destroy_tracked_urban_location = yes
					if = { limit = { has_variable = pie_city_underpopulated_months } remove_variable = pie_city_underpopulated_months }
					if = { limit = { has_variable = pie_town_underpopulated_months } remove_variable = pie_town_underpopulated_months }
				}
			}
			else = {
				if = { limit = { has_variable = pie_city_underpopulated_months } remove_variable = pie_city_underpopulated_months }
				if = { limit = { has_variable = pie_town_underpopulated_months } remove_variable = pie_town_underpopulated_months }
			}
		}
	}
}

pie_burghers_exchange_country_bonus_monthly = {
	effect = {
		set_variable = { name = pie_burghers_exchange_city_count value = 0 }
		every_owned_location = {
			if = {
				limit = {
					any_buildings_in_location = {
						building_type = building_type:pie_burghers_exchange_city
					}
				}
				owner = {
					change_variable = { name = pie_burghers_exchange_city_count add = 1 }
				}
			}
		}
		if = {
			limit = {
				var:pie_burghers_exchange_city_count > 0
			}
			add_country_modifier = {
				modifier = pie_burghers_exchange_city_bonus
				years = -1
				size = var:pie_burghers_exchange_city_count
				mode = replace
			}
		}
		else_if = {
			limit = {
				has_country_modifier = pie_burghers_exchange_city_bonus
			}
			remove_country_modifier = pie_burghers_exchange_city_bonus
		}
	}
}

pie_clergy_wealth_drain_monthly = {
	effect = {
		if = {
			limit = {
				country_has_estate = estate_type:clergy_estate
			}
			every_owned_location = {
				if = {
					limit = {
						any_buildings_in_location = {
							building_type = building_type:pie_clergy_parish_town
						}
					}
					owner = {
						add_gold_to_estate = {
							estate_type = estate_type:clergy_estate
							value = {
								value = prev.location_tax_base
								multiply = -2
								min = -20

pie_track_player_urban_origin = {
	effect = {
		if = {
			limit = {
				location_rank = location_rank:rural_settlement
			}
			pie_clear_urban_origin = yes
		}
		else_if = {
			limit = {
				location_rank = location_rank:town
				NOT = { pie_has_any_pending_urban_origin = yes }
			}
			if = {
				limit = {
					NOT = { pie_has_any_urban_origin = yes }
				}
				set_variable = { name = pie_origin_player_town value = yes }
			}
		}
		else_if = {
			limit = {
				location_rank = location_rank:city
				NOT = { pie_has_any_pending_urban_origin = yes }
			}
			if = {
				limit = { has_variable = pie_origin_nobles_town }
				remove_variable = pie_origin_nobles_town
				set_variable = { name = pie_origin_nobles_city value = yes }
			}
			else_if = {
				limit = { has_variable = pie_origin_clergy_town }
				remove_variable = pie_origin_clergy_town
				set_variable = { name = pie_origin_clergy_city value = yes }
			}
			else_if = {
				limit = { has_variable = pie_origin_burghers_town }
				remove_variable = pie_origin_burghers_town
				set_variable = { name = pie_origin_burghers_city value = yes }
			}
			else_if = {
				limit = { has_variable = pie_origin_player_town }
				remove_variable = pie_origin_player_town
				set_variable = { name = pie_origin_player_city value = yes }
			}
			else_if = {
				limit = {
					NOT = { pie_has_any_urban_origin = yes }
				}
				set_variable = { name = pie_origin_player_city value = yes }
			}
		}
	}
