# War Momentum - Hardcoded On Action Hooks
# Maps engine on_actions to mod-specific implementations in noicthy_on_actions.txt

on_annex = {
	on_actions = {
		noicthy_on_annex
	}
}

on_capital_moved = {
	on_actions = {
		noicthy_on_capital_moved
	}
}

# Auto-conquest tagging: marks locations taken by auto_conquer nations.

on_location_changed_owner = {
	on_actions = {
		noicthy_on_auto_conquest
	}
}

# War end: blanket cleanup of auto-conquest tags on both sides.

on_ending_war = {
	on_actions = {
		noicthy_on_war_end_cleanup
	}
}

# Bootstrap: initialize for countries already at war on new game.

on_game_start = {
	on_actions = {
		noicthy_on_game_start
	}
}
