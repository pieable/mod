namespace = pie_estate_urban_project

pie_estate_urban_project.1 = {
	type = country_event
	title = pie_estate_urban_project.1.title
	desc = pie_estate_urban_project.1.desc

	immediate = {
		event_illustration_estate_effect = {
			foreground = estate_type:nobles_estate
			background = estate_type:nobles_estate
		}
	}

	option = {
		name = pie_estate_urban_project.approve
		custom_tooltip = { text = pie_estate_urban_project.1.approve.tt }

		hidden_effect = {
			scope:target_location = { pie_complete_town_urbanization_project = yes }
		}
	}

	option = {
		name = pie_estate_urban_project.reject
		custom_tooltip = { text = pie_estate_urban_project.1.reject.tt }

		hidden_effect = {
			scope:target_location = { pie_clear_urbanization_project = yes }
			if = {
				limit = { country_has_estate = estate_type:nobles_estate }
				add_estate_satisfaction = { type = estate_type:nobles_estate value = -0.20 }
			}
		}
	}
}
