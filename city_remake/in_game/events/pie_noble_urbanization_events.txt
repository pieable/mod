namespace = pie_noble_urban

pie_noble_urban.1 = {
	type = country_event
	title = pie_noble_urban.1.title
	desc = pie_noble_urban.1.desc

	immediate = {
		event_illustration_estate_effect = {
			foreground = estate_type:nobles_estate
			background = estate_type:nobles_estate
		}
	}

	option = {
		name = pie_noble_urban.1.a
		custom_tooltip = { text = pie_noble_urban.1.a.tt }

		hidden_effect = {
			scope:target_location = {
				set_variable = { name = pie_pending_origin_nobles_town value = yes }
				pie_clear_urban_origin = yes
				change_location_rank = location_rank:town
				construct_estate_building = {
					building_type = building_type:pie_noble_manor_town
					estate_type = estate_type:nobles_estate
				}
				remove_variable = pie_pending_origin_nobles_town
				set_variable = { name = pie_origin_nobles_town value = yes }
			}

			if = {
				limit = {
					country_has_estate = estate_type:nobles_estate
				}
				add_gold_to_estate = {
					estate_type = estate_type:nobles_estate
					value = 700
				}
			}
		}
	}

	option = {
		name = pie_noble_urban.1.b
		custom_tooltip = { text = pie_noble_urban.1.b.tt }

		hidden_effect = {
			if = {
				limit = {
					country_has_estate = estate_type:nobles_estate
				}
				add_estate_satisfaction = { type = estate_type:nobles_estate value = -0.10 }
			}
		}
	}
}
